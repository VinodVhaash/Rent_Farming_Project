Farmer Table
CREATE TABLE Farmers
(
    FarmerId INT IDENTITY(1,1) PRIMARY KEY,

    UserId AS ('FA' + RIGHT('00000' + CAST(FarmerId AS VARCHAR(5)), 5)) PERSISTED UNIQUE,

    FirstName VARCHAR(50) NOT NULL,
    LastName VARCHAR(50) NOT NULL,

    Address VARCHAR(500) NOT NULL,

    SurveyNumber VARCHAR(100) NOT NULL,

    PhoneNumber VARCHAR(15) NOT NULL UNIQUE,

    PasswordHash VARCHAR(255) NOT NULL,

    CreatedDate DATETIME DEFAULT GETDATE(),

    IsActive BIT DEFAULT 1
);


Buyer Table
CREATE TABLE Buyers
(
    BuyerId INT IDENTITY(1,1) PRIMARY KEY,

    UserId AS ('BYR' + RIGHT('00000' + CAST(BuyerId AS VARCHAR(5)), 5)) PERSISTED UNIQUE,

    FirstName VARCHAR(50) NOT NULL,
    LastName VARCHAR(50) NOT NULL,

    Address VARCHAR(500) NOT NULL,

    PhoneNumber VARCHAR(15) NOT NULL UNIQUE,

    PasswordHash VARCHAR(255) NOT NULL,

    CreatedDate DATETIME DEFAULT GETDATE(),

    IsActive BIT DEFAULT 1
);
